☁ Google Sheets Integration

✅ All activities sync automatically. Tasks · Work Logs · Sites · PCB · Reassignments · Activities · HOD Sign-off
📋 HOD can delete rows directly from the Sheet using the Delete buttons in the HOD Audit panel.
📋 Setup Steps
1. Open your Google Sheet → Extensions → Apps Script
2. Delete default code. Paste the script below.
3. Click Deploy → New Deployment → Type: Web App
4. Set "Who has access" → "Anyone"
5. Deploy → copy the URL → paste above → Save & Connect
6. Click Sync All Data Now.
Complete Apps Script — supports delete by ID
function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); if (data.action === 'batchAppend') { data.batches.forEach(function(item) { var sh = getOrCreateSheet(ss, item.sheet); sh.appendRow(item.row); }); } if (data.action === 'append') { var sh = getOrCreateSheet(ss, data.sheet); sh.appendRow(data.row); } if (data.action === 'replace') { var sh = getOrCreateSheet(ss, data.sheet); sh.clearContents(); if (data.rows && data.rows.length > 0) { sh.getRange(1,1,data.rows.length,data.rows[0].length).setValues(data.rows); } } if (data.action === 'deleteRow') { var sh = getOrCreateSheet(ss, data.sheet); var vals = sh.getDataRange().getValues(); for (var i = 1; i < vals.length; i++) { if (String(vals[i][0]) === String(data.id)) { sh.deleteRow(i + 1); break; } } } return ContentService.createTextOutput( JSON.stringify({status:'OK'}) ).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput( JSON.stringify({status:'ERROR',msg:err.message}) ).setMimeType(ContentService.MimeType.JSON); } } function getOrCreateSheet(ss, name) { return ss.getSheetByName(name) || ss.insertSheet(name); } function doGet(e) { return ContentService.createTextOutput('Ecolyser API Active ✓'); }

1. Go to console.firebase.google.com → Create a project (free) 2. Click "Realtime Database" → Create Database → Start in TEST mode (Test mode allows read/write — fine for a private team tool) 3. Copy the URL shown (looks like: https://your-project-default-rtdb.firebaseio.com) Paste it above → Save & Connect → Push All Existing Data ✅ Done! Every device that opens this file will now share the same data. ⚠ Auto-syncs every 30 seconds. Data is also saved locally as backup.
